67      //字节代替
68      public char[] subBytes(char[] state) {
69          char[] result = new char[state.length];
70          for (int i = 0; i < state.length; i++) {
71              String s = Integer.toHexString(state[i]);
72              if (s.length() < 2) {
73                  s = "0" + s;
74              }
75              String rs = Sbox[s.charAt(0) < 97 ? s.charAt(0) - 48 : s.charAt(0) - 87][s.charAt(1) < 97 ? s.charAt(1) - 48 : s.charAt(1) - 87];
76              result[i] = (char) Integer.parseInt(rs, 16);
77          }
78          return result;
79      }

95      //列混淆
96      public char[] mixColumns(char[] state) {
97          char[] lisa = {2, 3, 1, 1};
98          char[] result = new char[16];
99          for (int col = 0; col < 4; col++) {
100             char[] lisb = new char[4];
101             int flagc = col;
102             for (int m = 0; m < 4; m++) {
103                 lisb[m] = state[flagc];
104                 flagc += 4;
105             }
106             for (int row = 0; row < 4; row++) {
107                 int k = ffmul(lisb[0], lisa[(4 - row) % 4]) ^ ffmul(lisb[1], lisa[(5 - row) % 4]) ^ ffmul(lisb[2], lisa[(6 - row) % 4]) ^ ffmul(lisb[3], lisa[(7 - row) % 4]);
108                 result[row * 4 + col] = (char) k;
109             }
110         }
111         return result;
112     }

133     //字节乘法
134     public int ffmul(int a, int b) {
135         String ba = Integer.toBinaryString(a);
136         int[] stor = new int[8];
137         while (ba.length() < 8) {
138             ba = "0" + ba;
139         }
140         stor[0] = b;
141         for (int i = 1; i < 8; i++) {
142             String bb = Integer.toBinaryString(stor[i - 1]);
143             if (bb.length() < 8) {
144                 stor[i] = leftshift(stor[i - 1], 1);
145             } else
146                 stor[i] = leftshift(stor[i - 1], 1) ^ 27;
147         }
148         int result = 0;
149         for (int i = 7; i >= 0; i--) {
150             if (ba.charAt(i) == '1') {
151                 if (result == 0) {
152                     result = stor[7 - i];
153                 } else
154                     result = result ^ stor[7 - i];
155             }
156         }
157         return result;
158     }

160     //二进制数左移
161     public int leftshift(int num, int step) {
162         String ba = Integer.toBinaryString(num);
163         while (ba.length() < 8) {
164             ba = "0" + ba;
165         }
166         for (int i = 0; i < step; i++) {
167             ba += "0";
168         }
169         return Integer.parseInt(ba.substring(step), 2);
170     }

172     //行移位
173     public char[] shiftRows(char[] state) {
174         char[][] in = new char[4][4];
175         for (int i = 0; i < 4; i++) {
176             for (int j = 0; j < 4; j++) {
177                 in[i][j] = state[i * 4 + j];
178             }
179         }
180         char[] result = new char[16];
181         for (int i = 0; i < 4; i++) {
182             for (int j = 0; j < 4; j++) {
183                 result[i * 4 + j] = in[i][(j + i) % 4];
184             }
185         }
186         return result;
187     }

189     //逆行移位
190     public char[] invShiftRows(char[] state) {
191         char[][] in = new char[4][4];
192         for (int i = 0; i < 4; i++) {
193             for (int j = 0; j < 4; j++) {
194                 in[i][j] = state[i * 4 + j];
195             }
196         }
197         char[] result = new char[16];
198         for (int i = 0; i < 4; i++) {
199             for (int j = 0; j < 4; j++) {
200                 int col = (j - i) > 0 ? (j - i) : (j - i) + 4;
201                 result[i * 4 + j] = in[i][col % 4];
202             }
203         }
204         return result;
205     }

207     //轮密钥加
208     public char[] addRoundKey(char[] state, char[] key) {
209         char[] result = new char[16];
210         for (int col = 0; col < 4; col++) {
211             for (int row = 0; row < 4; row++) {
212                 result[row * 4 + col] = (char) (state[row * 4 + col] ^ key[row * 4 + col]);
213             }
214         }
215         return result;
216     }

218     //密钥扩展
219     public char[][] keyExpansion(char[] key, int round) {
220         char[] RC = new char[round];
221         for (int i = 0; i < round; i++) {
222             if (i == 0) {
223                 RC[i] = 1;
224             } else {
225                 RC[i] = (char) ffmul(2, RC[i - 1]);
226             }
227         }
228 
229         char[][] newkey = new char[round][16];
230         char[] start = {key[12], key[13], key[14], key[15]};
231         for (int r = 0; r < round; r++) {
232             for (int i = 3; i < 7; i++) {
233                 if (i == 3) {
234                     char ot = start[0];
235                     start[0] = start[1];
236                     start[1] = start[2];
237                     start[2] = start[3];
238                     start[3] = ot;//RotWord()
239                     start = subBytes(start);
240                     start = XOR(start, new char[]{RC[r], 0, 0, 0});
241                 }
242                 char[] last = {key[(i - 3) * 4], key[(i - 3) * 4 + 1], key[(i - 3) * 4 + 2], key[(i - 3) * 4 + 3]};
243                 start = XOR(start, last);
244                 for (int k = 0; k < 4; k++) {
245                     key[(i - 3) * 4 + k] = start[k];
246                 }
247                 for (int j = 0; j < 4; j++) {
248                     newkey[r][(i + 1) % 4 + j * 4] = start[j];
249                 }
250             }
251         }
252         return newkey;
253     }

255     //异或
256     public char[] XOR(char[] a, char[] b) {
257         char[] result = new char[a.length];
258         for (int i = 0; i < a.length; i++) {
259             result[i] = (char) (((int) a[i]) ^ ((int) b[i]));
260         }
261         return result;
262     }

264     //字节加密
265     public char[] cipher(char[] in, char[] key, int round) {
266         char[] out = new char[16];
267         char[] newword = new char[16];
268         for (int i = 0; i < 16; i++) {
269             newword[i] = key[i];
270         }
271         char[][] keys = keyExpansion(newword, round);
272         in = exchange(in);
273         key = exchange(key);
274         in = addRoundKey(in, key);
275 
276         for (int i = 0; i < round - 1; i++) {
277             out = subBytes(in);
278             out = shiftRows(out);
279             out = mixColumns(out);
280             in = addRoundKey(out, keys[i]);
281         }
282         out = subBytes(in);
283         out = shiftRows(out);
284         out = addRoundKey(out, keys[round - 1]);
285         return exchange(out);
286     }

289     public String cipher(String in, String key, int round) {
290         StringBuffer sb = new StringBuffer();
291         char[] keys = Tools.hexStr2Cs(key);
292         String hexStr = Tools.Str2hexStr(in);
293         while (hexStr.length() >= 32) {
294             String sin = hexStr.substring(0, 32);
295             hexStr = hexStr.substring(32);
296             sb.append(Tools.Cs2hexStr(cipher(Tools.hexStr2Cs(sin), keys, round)));
297         }
298         if (hexStr.length() > 0) {
299             while (hexStr.length() < 32) {
300                 hexStr += "0";
301             }
302             sb.append(Tools.Cs2hexStr(cipher(Tools.hexStr2Cs(hexStr), keys, round)));
303         }
304         return sb.toString();
305     }

312     public char[] inCipher(char[] in, char[] key, int round) {
313         char[] out = new char[16];
314         char[] newword = new char[16];
315         for (int i = 0; i < 16; i++) {
316             newword[i] = key[i];
317         }
318         char[][] keys = keyExpansion(newword, round);
319         in = exchange(in);
320         key = exchange(key);
321         in = addRoundKey(in, keys[round - 1]);
322 
323         for (int i = 0; i < round - 1; i++) {
324             out = invShiftRows(in);
325             out = invSubBytes(out);
326             out = addRoundKey(out, keys[round - 2 - i]);
327             in = invMixColumns(out);
328 
329         }
330         out = invShiftRows(in);
331         out = invSubBytes(out);
332         out = addRoundKey(out, key);
333         return exchange(out);
334     }

336     //字符串解密
337     public String inCipher(String in, String key, int round) {
338         StringBuffer sb = new StringBuffer();
339         char[] keys = Tools.hexStr2Cs(key);
340         String hexStr = in;//Str2hexStr(in);
341         while (hexStr.length() >= 32) {
342             String sin = hexStr.substring(0, 32);
343             hexStr = hexStr.substring(32);
344             sb.append(Tools.hexStr2Str(Tools.Cs2hexStr(inCipher(Tools.hexStr2Cs(sin), keys, round))));
345         }
346         if (hexStr.length() > 0) {
347             while (hexStr.length() < 32) {
348                 hexStr += "0";
349             }
350             sb.append(Tools.hexStr2Str(Tools.Cs2hexStr(inCipher(Tools.hexStr2Cs(hexStr), keys, round))));
351         }
352         while (sb.charAt(sb.length() - 1) == 0) {
353             sb = sb.deleteCharAt(sb.length() - 1);
354         }
355         return sb.toString();
356     }

362     //行列变换
363     public char[] exchange(char[] chars) {
364         char[] nchars = new char[chars.length];
365         for (int i = 0; i < 4; i++) {
366             for (int j = 0; j < 4; j++) {
367                 nchars[i * 4 + j] = chars[j * 4 + i];
368             }
369         }
370         return nchars;
371     }
