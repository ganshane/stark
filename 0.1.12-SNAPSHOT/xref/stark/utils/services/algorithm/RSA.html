14  public class RSA {
15      KEY ku;//公钥
16      KEY kr;//私钥
17  
18      public RSA() {
19          Random r = new Random();
20          BigInteger p, q, n, nn;
21  
22          BigInteger e = BigInteger.ONE;// = new BigInteger(3+"");
23          BigInteger d = BigInteger.ONE;
24          //素数p,q,e,d
25          while (true) {
26              p = BigInteger.probablePrime(17, r);//new BigInteger(7+"");
27              q = BigInteger.probablePrime(19, r);//new BigInteger(5+"");
28  
29              n = p.multiply(q);
30  
31              nn = p.subtract(BigInteger.ONE).multiply(q.subtract(BigInteger.ONE));
32              if (nn.longValue() > 65535) {
33  
34                  for (int i = 3; i < nn.intValue(); i++) {
35                      if (MyMath.gcd(i, nn.intValue()) == 1) {
36                          e = BigInteger.valueOf(i);
37  
38                          if (MyMath.exgcd(e, nn).longValue() == -1)
39                              continue;
40                          else
41                              break;
42                      }
43                  }
44                  d = MyMath.exgcd(e, nn).mod(nn);
45                  BigInteger big = d.multiply(d);
46                  big = big.multiply(big);
47                  if (big.compareTo(n) > 0) {
48                      break;
49                  } else
50                      continue;
51              }
52  
53          }
54  
55          this.ku = new KEY(e, n);
56          this.kr = new KEY(d, n);
57      }
58  
59      public RSA(KEY ku, KEY kr) {
60          super();
61          this.ku = ku;
62          this.kr = kr;
63      }
64  
65      class KEY {
66          BigInteger x;
67          BigInteger n;
68  
69          public KEY(BigInteger x, BigInteger n) {
70              super();
71              this.x = x;
72              this.n = n;
73          }
74  
75      }
76  
77      //加密
78      public String Encryption(String s, KEY key) {
79          StringBuffer sb = new StringBuffer();
80          char[] cs = s.toCharArray();
81          for (int i = 0; i < cs.length; i++) {
82              int k = cs[i];
83              if (k < key.n.longValue()) {
84                  BigInteger p = new BigInteger(k + "");
85                  int kk = Integer.parseInt(MyMath.reaminder(key.n, p, Long.parseLong(key.x.toString())).toString());
86                  char c = (char) kk;
87                  sb.append(c);
88              } else {
89                  sb.append(cs[i]);
90              }
91          }
92          return sb.toString();
93      }
94  
95      //加密成16进制字符串
96      public String Encryption(String s) {
97          StringBuffer sb = new StringBuffer();
98          char[] cs = s.toCharArray();
99          for (int i = 0; i < cs.length; i++) {
100             int k = cs[i];
101             if (k < this.ku.n.longValue()) {
102                 BigInteger p = new BigInteger(k + "");
103                 long kk = Long.parseLong(MyMath.reaminder(this.ku.n, p, Long.parseLong(this.ku.x.toString())).toString());
104                 sb.append(Long.toHexString(kk));
105                 sb.append(" ");
106             } else {
107                 sb.append(Integer.toHexString(k));
108                 sb.append(" ");
109             }
110         }
111         return sb.toString();
112     }
113 
114     //解密
115     public String Decryption(String s, KEY key) {
116         return Encryption(s, key);
117     }
118 
119     public String Decryption(String s) {
120         StringBuffer sb = new StringBuffer();
121         String[] ss = s.split(" ");
122         for (int i = 0; i < ss.length; i++) {
123             long k = Long.parseLong(ss[i], 16);
124             BigInteger p = new BigInteger(k + "");
125             int kk = Integer.parseInt(MyMath.reaminder(this.kr.n, p, Long.parseLong(this.kr.x.toString())).toString());
126             sb.append(Tools.obox(Integer.toHexString(kk), 4));
127         }
128         return Tools.hexStr2Str(sb.toString());
129     }
