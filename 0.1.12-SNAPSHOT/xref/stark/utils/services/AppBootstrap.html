3   package stark.utils.services;
4   
5   import java.io.File;
6   import java.lang.reflect.Method;
7   import java.net.URL;
8   import java.net.URLClassLoader;
9   import java.util.ArrayList;
10  import java.util.List;
11  
12  /**
13   * application bootstrap
14   */
15  public class AppBootstrap {
16      protected static void start(String mainClassName, String[] args) throws Exception {
17          //get server home
18          String serverHome = System.getProperty("server.home", "support");
19          String LIB = serverHome + File.separator + "lib";
20  
21          List<URL> urls = new ArrayList<URL>();
22          File[] files = new File(LIB).listFiles();
23  
24          if (files != null) {
25              for (File f : files) {
26                  urls.add(f.toURI().toURL());
27              }
28          }
29          // feed your URLs to a URLClassLoader!
30          ClassLoader classloader = new URLClassLoader(
31                  urls.toArray(new URL[urls.size()]),
32                  ClassLoader.getSystemClassLoader().getParent());
33  
34          // well-behaved Java packages work relative to the
35          // context classloader.  Others don't (like commons-logging)
36          Thread.currentThread().setContextClassLoader(classloader);
37  
38          // relative to that classloader, find the main class
39          // you want to bootstrap, which is the first cmd line arg
40          Class mainClass = classloader.loadClass(mainClassName);
41          Method main = mainClass.getMethod("main",
42                  new Class[]{args.getClass()});
43  
44          main.invoke(null, new Object[]{args});
45      }
46  }
