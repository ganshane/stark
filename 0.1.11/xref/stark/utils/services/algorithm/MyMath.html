72      public static boolean isPrime(long num) {
73          boolean flag = true;
74          for (long i = 2; i < num / 2; i++) {
75              if (num == 2) break;
76              if (num % i == 0) {
77                  flag = false;
78                  break;
79              }
80          }
81          return flag;
82      }

95      public static long exgcd(long a, long b) {
96          long x1 = 1, x2 = 0, x3 = b,
97                  y1 = 0, y2 = 1, y3 = a;
98          while (true) {
99              if (y3 == 0) {
100                 return -1;
101             }
102             if (y3 == 1) {
103                 return y2 > 0 ? y2 : y2 + b;
104             }
105             long t1, t2, t3;
106             long q = x3 / y3;
107             t1 = x1 - q * y1;
108             t2 = x2 - q * y2;
109             t3 = x3 - q * y3;
110             x1 = y1;
111             x2 = y2;
112             x3 = y3;
113             y1 = t1;
114             y2 = t2;
115             y3 = t3;

120         BigInteger x1 = BigInteger.ONE, x2 = BigInteger.ZERO, x3 = b,
121                 y1 = BigInteger.ZERO, y2 = BigInteger.ONE, y3 = a;
122         while (true) {
123             if (y3.equals(BigInteger.ZERO)) {
124                 return BigInteger.ZERO.subtract(BigInteger.ONE);
125             }
126             if (y3.equals(BigInteger.ONE)) {
127                 return y2;
128             }
129             BigInteger t1, t2, t3;
130             BigInteger q = x3.divide(y3);//x3/y3;
131             t1 = x1.subtract(q.multiply(y1));//x1-q*y1;
132             t2 = x2.subtract(q.multiply(y2));//x2 - q*y2;
133             t3 = x3.subtract(q.multiply(y3));//x3 - q*y3;
134             x1 = y1;
135             x2 = y2;
136             x3 = y3;
137             y1 = t1;
138             y2 = t2;
139             y3 = t3;
140         }
